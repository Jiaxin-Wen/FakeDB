SELECT NATION.N_NAME, SUPPLIER.S_NAME FROM NATION, SUPPLIER WHERE NATION.N_NATIONKEY=SUPPLIER.S_NATIONKEY;
SELECT NATION.N_NAME, SUPPLIER.S_NAME FROM NATION, SUPPLIER WHERE NATION.N_NATIONKEY=SUPPLIER.S_NATIONKEY and NATION.NAME='CHINA';
SELECT REGION.R_NAME, NATION.N_NAME, SUPPLIER.S_NAME FROM REGION, NATION, SUPPLIER WHERE REGION.R_REGIONKEY=NATION.N_NATIONKEY and NATION.N_NATIONKEY=SUPPLIER.S_NATIONKEY and NATION.NAME='CHINA';
