SELECT NATION.N_NAME, SUPPLIER.S_NAME FROM NATION, SUPPLIER WHERE NATION.N_NATIONKEY=SUPPLIER.S_NATIONKEY;
SELECT NATION.N_NAME, SUPPLIER.S_NAME FROM NATION, SUPPLIER WHERE NATION.N_NATIONKEY=SUPPLIER.S_NATIONKEY AND NATION.N_NAME='CHINA';
SELECT REGION.R_NAME, NATION.N_NAME, SUPPLIER.S_NAME FROM REGION, NATION, SUPPLIER WHERE REGION.R_REGIONKEY=NATION.N_REGIONKEY AND NATION.N_NATIONKEY=SUPPLIER.S_NATIONKEY AND NATION.N_NAME='CHINA';
