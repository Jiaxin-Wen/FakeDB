CREATE TABLE student(
    name VARCHAR(25) NOT NULL,
    class_id INT NOT NULL
);
CREATE TABLE class(
    id INT NOT NULL,
    name VARCHAR(25) NOT NULL
);
CREATE TABLE teacher(
    name VARCHAR(25) NOT NULL,
    class_id INT NOT NULL
);
